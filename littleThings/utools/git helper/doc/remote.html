$ git fetch [remote]
注解 # 下载远程仓库的所有变动
$ git remote -v
注解 # 显示所有远程仓库
$ git remote show [remote]
注解 # 显示某个远程仓库的信息
$ git remote add [shortname] [url]
注解 # 增加一个新的远程仓库，并命名
$ git remote -v
注解 # 查看远程服务器地址和仓库名称
$ git remote add origin git@ github:xxx/xxx.git
注解 # 添加远程仓库地址
$ git remote set-url origin git@ github.com:xxx/xxx.git
注解 # 设置远程仓库地址(用于修改远程仓库地址)
$ git remote rm
注解 # 删除远程仓库
