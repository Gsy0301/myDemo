$ git tag
注解 # 列出所有tag
$ git tag [tag]
注解 # 新建一个tag在当前commit
$ git tag [tag] [commit]
注解 # 新建一个tag在指定commit
$ git tag -d [tag]
注解 # 删除本地tag
$ git push origin :refs/tags/[tagName]
注解 # 删除远程tag
$ git show [tag]
注解 # 查看tag信息
$ git push [remote] [tag]
注解 # 提交指定tag
$ git push [remote] --tags
注解 # 提交所有tag
$ git checkout -b [branch] [tag]
注解 # 新建一个分支，指向某个tag
