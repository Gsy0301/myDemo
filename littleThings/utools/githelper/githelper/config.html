# 显示当前的Git配置

$ git config --list

# 编辑Git配置文件

$ git config -e [--global]

# 输出、设置基本的全局变量

$ git config --global user.email

$ git config --global user.name

$ git config --global user.email "MyEmail@gmail.com"

$ git config --global user.name "My Name"

# 定义当前用户所有提交使用的作者邮箱。

$ git config --global alias.

# 为Git命令创建一个快捷方式（别名）。

$ git config --system core.editor
